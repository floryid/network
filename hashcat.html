Hashcat

Perintah Dasar

# Jalankan Hashcat dengan pengaturan default
hashcat -m [format] -a [mode] [file_hash] [wordlist]

# Contoh:
hashcat -m 0 -a 0 hashes.txt wordlist.txt

Format Hash Umum

# Format hash yang umum digunakan:
- 0: MD5
- 100: SHA1
- 1400: SHA256
- 22000: WPA/WPA2
- 2600: bcrypt

Mode Serangan

# Jenis mode serangan:
- 0: Straight (Wordlist)
- 1: Combination
- 3: Brute-force
- 6: Hybrid Wordlist + Mask
- 7: Hybrid Mask + Wordlist

Contoh Serangan dengan Wordlist

# Gunakan wordlist untuk menyerang hash
hashcat -m 0 -a 0 hashes.txt wordlist.txt

Brute-force dengan Mask

# Serangan brute-force menggunakan mask
hashcat -m 0 -a 3 hashes.txt ?l?l?l?l?l?l

# Contoh:
hashcat -m 0 -a 3 hashes.txt ?u?u?u?u?u?u (6 huruf kapital)

Menampilkan Hasil

# Menampilkan hasil setelah proses cracking
hashcat --show hashes.txt

Menyimpan dan Mengambil Sesi

# Menyimpan sesi yang sedang berjalan
hashcat -m [format] -a [mode] --session=my_session hashes.txt wordlist.txt

# Mengambil sesi yang disimpan
hashcat --restore

Melihat Format Hash yang Didukung

# Untuk melihat semua format hash yang didukung
hashcat -h | grep -A 10 "Supported hash types"

Opsi Tambahan

# Menggunakan aturan pada wordlist
hashcat -m 0 -a 0 -r rules/best64.rule hashes.txt wordlist.txt

Pengaturan GPU

# Untuk mengatur penggunaan GPU
hashcat -m 0 -a 0 --opencl-device-types 1 hashes.txt wordlist.txt
